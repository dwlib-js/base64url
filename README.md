# Base64URL

## Install
`npm i --save @dwlib/base64url`

## Usage
```javascript
// CJS
const Base64URL = require('@dwlib/base64url');
const Base64URLDecode = require('@dwlib/base64url/decode');
const Base64URLDecodeInto = require('@dwlib/base64url/decodeInto');
const Base64URLEncode = require('@dwlib/base64url/encode');
const Base64URLIsValid = require('@dwlib/base64url/isValid');
const Base64URLNormalize = require('@dwlib/base64url/normalize');
const Base64URLValidate = require('@dwlib/base64url/validate');
// ESM
import Base64URL from '@dwlib/base64url';
import Base64URLDecode from '@dwlib/base64url/decode';
import Base64URLDecodeInto from '@dwlib/base64url/decodeInto';
import Base64URLEncode from '@dwlib/base64url/encode';
import Base64URLIsValid from '@dwlib/base64url/isValid';
import Base64URLNormalize from '@dwlib/base64url/normalize';
import Base64URLValidate from '@dwlib/base64url/validate';
```
